const express = require('express'); const path = require('path'); const app = express(); // Middleware app.use(express.static('public')); app.use(express.json()); // Store projects in memory const projects = []; // Main page app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'index.html')); }); // Create new project app.post('/api/projects', (req, res) => { const { name, type } = req.body; const project = { id: Date.now(), name, type, created: new Date().toISOString() }; projects.push(project); res.json(project); }); // Get all projects app.get('/api/projects', (req, res) => { res.json(projects); }); // Start server const port = process.env.PORT || 3000; app.listen(port, () => { console.log(`Server running on port ${port}`); });
